```typescript
{IMAGE_MODELS.map(model => {
  // 如果上传了图片，只显示支持编辑的模型
  if (uploadedImage && !model.supportsImageEdit) {
    return null;
  }
  
  // 如果没有上传图片，显示所有模型
  // 如果上传了图片，只显示支持编辑的模型（上面已过滤）
  
  return (
    <div key={model.id} className={`model-card ...`}>
      {/* 模型卡片内容 */}
    </div>
  );
})}
```
